server {
    listen 443 http2 ssl;

    server_name dev-user-media.coucher.org;

    access_log /var/log/nginx/dev-user-media.coucher.org.access.log;
    error_log /var/log/nginx/dev-user-media.coucher.org.error.log;

    ssl_certificate /certs/live/dev-user-media.coucher.org/fullchain.pem;
    ssl_certificate_key /certs/live/dev-user-media.coucher.org/privkey.pem;

    location /media/ {
        proxy_http_version 1.1;
        proxy_pass http://media:5000/;
    }

    location = / {
        add_header Content-Type text/plain;

        set $msg "This is the media server for the Couchers.org app. If you're interested in helping out in building";
        set $msg "${msg} the next-generation couch-surfing platform with us, please contact us through";
        set $msg "${msg} https://couchers.org or come help out on GitHub at https://github.com/Couchers-org.\n";

        return 200 $msg;
    }

    location / {
        return 404;
    }
}
